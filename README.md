# letstestit

//installation Main App
1. composer install

2. php artisan key:generate and php artisan jwt:secret

3. php artisan migrate

4. php artisan custom_storage:link

5. npm install


//installation Admin
1. composer install

2. php artisan key:generate and php artisan jwt:secret

3. php artisan migrate

4. php artisan storage:link

5. npm install

//usage Main App
1. npm run dev

//usage Admin
1. npm run watch
